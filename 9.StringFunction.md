## 🔹 String Functions

---

## 📌 LENGTH()

```sql id="sf1"
-- Returns number of characters in a string
SELECT LENGTH('skyfall');
```

---

```sql id="sf2"
-- Check length of each first name
SELECT first_name, LENGTH(first_name)
FROM employee_demographics;
```

---

```sql id="sf3"
-- Order by length (using column position)
SELECT first_name, LENGTH(first_name)
FROM employee_demographics
ORDER BY 2;
```

---

```sql id="sf4"
-- Order by length using function
SELECT first_name, LENGTH(first_name)
FROM employee_demographics
ORDER BY LENGTH(first_name);
```

---

```sql id="sf5"
-- Using alias for readability
SELECT first_name, LENGTH(first_name) AS length_of_name
FROM employee_demographics
ORDER BY length_of_name DESC;
```

## 🔹 UPPER() & LOWER()

```sql id="sf6"
-- Convert to uppercase
SELECT UPPER('Saimmi');
```

```sql id="sf7"
SELECT first_name, UPPER(first_name) AS uppercase_name
FROM employee_demographics;
```

---

```sql id="sf8"
-- Convert to lowercase
SELECT LOWER('SAIMMI');
```

```sql id="sf9"
SELECT first_name, LOWER(first_name) AS lowercase_name
FROM employee_demographics;
```

---

## 🔹 TRIM Functions

```sql id="sf10"
-- Remove spaces from both sides
SELECT TRIM('    Saimmi    ');
```

```sql id="sf11"
-- Remove spaces from left side
SELECT LTRIM('    Saimmi    ');
```

```sql id="sf12"
-- Remove spaces from right side
SELECT RTRIM('    Saimmi    ');
```

---

## 🔹 LEFT() & RIGHT()

```sql id="sf13"
-- Get first 4 characters
SELECT first_name, LEFT(first_name, 4)
FROM employee_demographics;
```

---

```sql id="sf14"
-- Get last 4 characters
SELECT last_name, RIGHT(last_name, 4)
FROM employee_demographics;
```

---

```sql id="sf15"
-- Combine LEFT and RIGHT
SELECT first_name,
       LEFT(first_name, 4),
       RIGHT(first_name, 4)
FROM employee_demographics;
```

---

```sql id="sf16"
SELECT first_name,
       last_name,
       LEFT(first_name, 3),
       RIGHT(last_name, 3)
FROM employee_demographics;
```

---

## 🔹 SUBSTRING()

```sql id="sf17"
-- Extract substring (start position, length)
SELECT first_name, SUBSTRING(first_name, 3, 2)
FROM employee_demographics;
```

---

```sql id="sf18"
-- Extract month from date
SELECT birth_date,
       SUBSTRING(birth_date, 6, 2) AS birth_month
FROM employee_demographics;
```

---

## 🔹 REPLACE()

```sql id="sf19"
-- Replace characters in string
SELECT first_name,
       REPLACE(first_name, 'a', 'z')
FROM employee_demographics;
```

---

## 🔹 LOCATE()

```sql id="sf20"
-- Find position of character
SELECT LOCATE('x', 'Alexander');
```

```sql id="sf21"
SELECT LOCATE('x', 'Saimmi');
```

```sql id="sf22"
SELECT LOCATE('m', 'saimmi');
```

---

```sql id="sf23"
-- Find position in column values
SELECT first_name,
       LOCATE('a', first_name)
FROM employee_demographics;
```

---

## 🔹 CONCAT()

```sql id="sf24"
-- Combine first and last name
SELECT first_name,
       last_name,
       CONCAT(first_name, last_name)
FROM employee_demographics;
```

---

```sql id="sf25"
-- Create custom short name
SELECT first_name,
       last_name,
       LEFT(first_name, 3) AS first_n,
       RIGHT(last_name, 3) AS last_n,
       CONCAT(LEFT(first_name, 3), RIGHT(last_name, 3)) AS full_short
FROM employee_demographics;
```

---

## 🧠 Key Notes

* `LENGTH()` → counts characters
* `UPPER()` / `LOWER()` → change case
* `TRIM()` → removes extra spaces
* `LEFT()` / `RIGHT()` → extract characters
* `SUBSTRING()` → extract part of string
* `REPLACE()` → replace text
* `LOCATE()` → find position of character
* `CONCAT()` → combine strings

---

## 🎯 Simple Understanding

* String functions help **clean, format, and manipulate text data**
