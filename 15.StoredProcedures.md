## 🔹 Stored Procedures

---

## 📌 Basic Queries

```sql id="sp_a1"
-- View all employee salaries
SELECT *
FROM employee_salary;
```

```sql id="sp_a2"
-- Filter employees with salary >= 50000
SELECT *
FROM employee_salary
WHERE salary >= 50000;
```

---

## 📌 Create Simple Stored Procedure

```sql id="sp_a3"
-- Create a procedure to get employees with salary >= 50000
CREATE PROCEDURE large_salaries()
SELECT *
FROM employee_salary
WHERE salary >= 50000;
```

```sql id="sp_a4"
-- Execute procedure
CALL large_salaries;
```

---

## 📌 Stored Procedure with Multiple Queries

```sql id="sp_a5"
DELIMITER $$

CREATE PROCEDURE large_salaries2()
BEGIN
    -- Employees with salary >= 50000
    SELECT *
    FROM employee_salary
    WHERE salary >= 50000;

    -- Employees with salary >= 10000
    SELECT *
    FROM employee_salary
    WHERE salary >= 10000;
END $$

DELIMITER ;
```

---

```sql id="sp_a6"
-- Same structure with proper execution
DELIMITER $$

CREATE PROCEDURE large_salaries3()
BEGIN
    SELECT *
    FROM employee_salary
    WHERE salary >= 50000;

    SELECT *
    FROM employee_salary
    WHERE salary >= 10000;
END $$

DELIMITER ;
```

```sql id="sp_a7"
CALL large_salaries3();
```

## 📌 Stored Procedure with Parameter

```sql id="sp_a8"
USE parks_and_recreation;

-- Drop procedure if it already exists
DROP PROCEDURE IF EXISTS large_salaries4;

DELIMITER $$

CREATE PROCEDURE large_salaries4(IN employee_id_param INT)
BEGIN
    -- Get salary of specific employee
    SELECT salary
    FROM employee_salary
    WHERE employee_id = employee_id_param;
END $$

DELIMITER ;
```

```sql id="sp_a9"
-- Call procedure with parameter
CALL large_salaries4(1);
```

## 📌 Stored Procedure with Input Parameter

```sql id="sp_a14"
DROP PROCEDURE IF EXISTS get_employee_by_id;

DELIMITER $$

CREATE PROCEDURE get_employee_by_id(IN emp_id INT)
BEGIN
    -- Get employee details by ID
    SELECT *
    FROM employee_demographics
    WHERE employee_id = emp_id;
END $$

DELIMITER ;
```

```sql id="sp_a15"
CALL get_employee_by_id(7);
```

---

```sql id="sp_a16"
DROP PROCEDURE IF EXISTS get_employee_by_gender;

DELIMITER $$

CREATE PROCEDURE get_employee_by_gender(IN emp_gender VARCHAR(10))
BEGIN 
    -- Get employees based on gender
    SELECT *
    FROM employee_demographics
    WHERE gender = emp_gender;
END $$

DELIMITER ;
```

```sql id="sp_a17"
CALL get_employee_by_gender('Female');
```

## 📌 Stored Procedure with JOIN

```sql id="sp_a18"
DELIMITER $$

CREATE PROCEDURE employee_salary_details()
BEGIN
    -- Combine employee name with salary
    SELECT 
        dem.first_name,
        dem.last_name,
        sal.salary
    FROM employee_demographics dem
    JOIN employee_salary sal
      ON dem.employee_id = sal.employee_id;
END $$

DELIMITER ;
```

```sql id="sp_a19"
CALL employee_salary_details();
```

---

## 🧠 Key Notes

* Stored procedures are reusable SQL blocks
* Executed using `CALL`
* `DELIMITER $$` is used to define procedure block
* `IN` parameter allows passing values
* Can contain multiple SQL statements
* Useful for:

  * Reusability
  * Reducing repeated queries
  * Organizing business logic
